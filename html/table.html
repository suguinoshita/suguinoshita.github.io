Circular $$ {\begin{align}x_{c} &= D/2 \\y_{c} &= D/2\end{align}} $$ $$ {A = \frac{\pi D^2}{4} = \pi R^2} $$ $$ {\begin{align}P &= \pi D \\&= 2 \pi R\end{align}} $$ $$ {I_{xx}=\frac{\pi D^4}{64}=\frac{\pi R^4}{4}} $$ $$ {I_{yy}=I_{xx}} $$ $$ {\begin{align}I_{zz} &= I_{xx} + I_{yy} \\ &= \frac{\pi D^4}{32}=\frac{\pi R^4}{2}\end{align}} $$ $$ {S_{x,min}=\frac{I_{xx}}{R}=\frac{\pi R^3}{4}} $$ $$ {S_{y,min}=S_{x,min}} $$
Circular tube $$ {\begin{align}x_{c} &= D/2 \\y_{c} &= D/2\end{align}} $$ $$ {\begin{align}A &= \frac{\pi (D_{o}^2-D_{i}^2)}{4} \\&= \pi (R_{o}^2-R_{i}^2)\end{align}} $$ $$ {\begin{align}P_{out} &= \pi D_{o} \\&= 2 \pi R_{o}\\P_{int} &= \pi D_{i} \\&= 2 \pi.R_{i}\end{align}} $$ $$ {\begin{align}I_{xx}&=\frac{\pi (D_{o}^4-D_{i}^4)}{64}\\&=\frac{\pi (R_{o}^4-R_{i}^4)}{4}\end{align}} $$ $$ {I_{yy}=I_{xx}} $$ $$ {\begin{align}I_{zz} &= I_{xx} + I_{yy} \\ &= \frac{\pi (D_{o}^4-D_{i}^4)}{32}\\&=\frac{\pi (R_{o}^4-R_{i}^4)}{2}\end{align}} $$ $$ {\begin{align}S_{x,min}&=\frac{I_{xx}}{R}\\&=\frac{\pi (R_{o}^4-R_{i}^4)}{4R_{o}}\end{align}} $$ $$ {S_{y,min}=S_{x,min}} $$
Rectangular $$ {\begin{align}x_{c} &= b/2 \\y_{c} &= h/2\end{align}} $$ $$ {A = h b} $$ $$ {P = 2 h + 2 b} $$ $$ {I_{xx}=\frac{bh^3}{12}} $$ $$ {I_{yy}=\frac{hb^3}{12}} $$ $$ {\begin{align}I_{zz} &= I_{xx} + I_{yy} \\ &= \frac{bh^3+hb^3}{12}\end{align}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}}=\frac{2 I_{yy}}{b}} $$
Rectangular tube
$$ {\begin{align}h_{i} &= h-2t\\b_{i}&=b-2t\end{align}} $$		 $$ {\begin{align}x_{c} &= b/2 \\y_{c} &= h/2\end{align}} $$ $$ {\begin{align}A &= hb - h_{i}b_{i}\\A &= 2(h+b)t - 4t^2\end{align}} $$ $$ {\begin{align}P_{out} &= 2 h + 2 b \\P_{int} &= 2(h+b)-8t\end{align}} $$ $$ {I_{xx}=\frac{bh^3-b_{i}h_{i}^3}{12}} $$ $$ {I_{yy}=\frac{hb^3-h_{i}b_{i}^3}{12}} $$ $$ {\begin{align}I_{zz} &= I_{xx} + I_{yy} \\ &=\frac{bh^3+hb^3-b_{i}h_{i}^3-h_{i}b_{i}^3}{12}\end{align}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}}=\frac{2 I_{yy}}{b}} $$
L section $$ {\begin{align}x_{c} &= \frac{t}{2A}(b^2+ht-t^2) \\y_{c} &= \frac{t}{2A}(h^2+bt-t^2)\end{align}} $$ $$ {A=(h+b)t-t^2} $$ $$ {P = 2 h + 2 b} $$ $$ {I_{xx}=\frac{(bt^2+h^3-t^3)t}{3}-A y_{c}^2} $$ $$ {I_{yy}=\frac{(ht^2+b^3-t^3)t}{3}-A x_{c}^2} $$ $$ {\begin{align}I_{zz} &= I_{xx} + I_{yy} \\ &=\frac{(h^3+b^3+(h+b)t^2-2 t^3)t}{3}-A(x_{c}^2+y_{c}^2)\end{align}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}}} $$
C channel $$ {\begin{align}x_{c} &= \frac{1}{A}(\frac{t_{w}^2}{2}(h-2 t_f)+t_f b^2 ) \\y_{c} &= h/2\end{align}} $$ $$ {A = 2 b t_f + (h-2 t_f)t_w} $$ $$ {P = 4b+2h-2t_w} $$ $$ {I_{xx}=\frac{bh^3}{12} - \frac{(b-t_w)(h-2t_f)^3}{12}} $$ $$ {I_{yy}=\frac{(h-2t_f)t_{w}^3+2t_{f}b^3}{3} - A x_{c}^2} $$ $$ {I_{zz} = I_{xx} + I_{yy}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}} } $$
U channel $$ {\begin{align}x_{c} &= \frac{1}{A}(\frac{t_{w}^2}{2}(b-2 t_f)+t_f h^2 ) \\y_{c} &= b/2\end{align}} $$ $$ {A = 2 h t_f + (b-2 t_f)t_w} $$ $$ {P = 4h+2b-2t_w} $$ $$ {I_{xx}=\frac{bh^3}{12} - \frac{(b-t_w)(h-2t_f)^3}{12}} $$ $$ {I_{yy}=\frac{(h-2t_f)t_{w}^3+2t_{f}b^3}{3} - A x_{c}^2} $$ $$ {I_{zz} = I_{xx} + I_{yy}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}} } $$
I section $$ {\begin{align}x_{c} &= b/2 \\y_{c} &= h/2\end{align}} $$ $$ {A = 2 b t_f + (h-2 t_f)t_w} $$ $$ {P = 4b+2h-2t_w} $$ $$ {I_{xx}=\frac{bh^3}{12} - \frac{(b-t_w)(h-2t_f)^3}{12}} $$ $$ {I_{yy}=\frac{2 t_{f} b^3 + (h-2 t_{f}) t_{w}^3}{12}} $$ $$ {I_{zz} = I_{xx} + I_{yy}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}}=\frac{2 I_{yy}}{b} } $$
H section $$ {\begin{align}x_{c} &= b/2 \\y_{c} &= h/2\end{align}} $$ $$ {A = 2 h t_f + (b-2 t_f)t_w} $$ $$ {P = 4b+2h-2t_w} $$ $$ {I_{xx}=\frac{2 t_{f} b^3 + (b-2 t_{f}) t_{w}^3}{12}} $$ $$ {I_{yy}=\frac{hb^3}{12} - \frac{(h-t_w)(b-2t_f)^3}{12}} $$ $$ {I_{zz} = I_{xx} + I_{yy}} $$ $$ {S_{x,min}=\frac{I_{xx}}{h - y_{c}}=\frac{2 I_{xx}}{h}} $$ $$ {S_{y,min}=\frac{I_{yy}}{b - x_{c}}=\frac{2 I_{yy}}{b}} $$
